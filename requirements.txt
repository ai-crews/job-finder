# ==== 📡 크롤링 및 파싱 관련 패키지 ====
requests           # HTTP 요청을 보낼 때 사용 (웹 페이지 가져오기 등)
beautifulsoup4     # HTML 파싱 라이브러리 (정적인 웹사이트 구조 해석)
lxml               # BeautifulSoup이 HTML을 빠르게 파싱하도록 도와줌 (선택적 성능 향상용)
selenium           # 동적인 웹사이트 크롤링 (예: 자바스크립트로 렌더링되는 페이지)

# ==== 🧹 데이터 처리 및 정리 ====
pandas             # 표 형태의 데이터 정리, 필터링, 저장 등에 사용

# ==== 🕒 스케줄링 및 자동화 ====
schedule           # 파이썬 내에서 코드 반복 실행 예약 (ex: 매주 월요일 9시)
croniter           # cron 형식 문자열을 파싱하고 계산 (스케줄 고도화할 때 사용)

# ==== ✉️ 이메일 발송 ====
yagmail            # Gmail 기반으로 이메일을 간편하게 전송할 수 있는 패키지
jinja2             # 이메일 HTML 템플릿을 변수에 따라 동적으로 채우기 (ex: 사용자 이름, 채용공고 리스트 등)

# ==== 📦 환경변수 및 설정 파일 관리 ====
python-dotenv      # .env 파일에서 API KEY, 비밀번호 같은 민감정보 읽을 수 있게 함

# ==== 📋 로깅 및 디버깅 ====
loguru             # 예쁘고 간편한 로그 기록 패키지 (디버깅과 운영 시 필수)

# ==== 🧪 테스트 ====
pytest             # 코드 테스트 자동화 프레임워크

# ==== ☁️ 서버 및 API 배포 (선택) ====
fastapi            # 빠르고 간단하게 API 서버 만들 수 있는 파이썬 프레임워크
uvicorn            # FastAPI 서버 실행용 lightweight ASGI 서버

# ==== 🗄️ 데이터베이스 연동 (옵션) ====
sqlalchemy         # 파이썬 ORM 라이브러리 (DB와 연동할 때 사용)
pymysql            # MySQL과 연결할 수 있는 드라이버
